# Frontend Redesign Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Redesign the Appalachian Actors Photoshare Flask app with a clean modern aesthetic, justified-row photo gallery, keyboard-navigable lightbox, 4-theme CSS system, JS linting/minification, and hardened CI.

**Architecture:** Pure frontend redesign — rewrite all templates and CSS, add two vanilla JS files, with minimal targeted backend additions (photo counts, video detection, pagination size). No new routes, no framework, no build-time dependencies in the Docker image.

**Tech Stack:** Flask/Jinja2, vanilla CSS (custom properties), vanilla JS (ES2022, no modules), ESLint 9 (flat config), esbuild for minification, GitHub Actions.

**Spec:** `docs/superpowers/specs/2026-03-17-frontend-redesign.md`

- [ ] **Step 3: Add `_count_photos` helper to app.py**

  Add after the existing helpers block (after `_list_directory_cached`, before decorators):

  ```python
  _IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png"}


  def _count_photos(show_rel: Path) -> int:
      """Count image files one level deep in *show_rel*, excluding Video/ subfolders.

      *show_rel* is a path relative to _MEDIA_ROOT.
      """
      show_path = _MEDIA_ROOT / show_rel
      total = 0
      try:
          for entry in show_path.iterdir():
              if entry.is_dir() and entry.name.lower() != "video":
                  total += sum(
                      1 for f in entry.iterdir()
                      if f.is_file() and f.suffix.lower() in _IMAGE_EXTENSIONS
                  )
      except FileNotFoundError:
          pass
      return total
  ```

  ```python
  @app.route("/directory/<path:directory>/<int:page>/")
  @login_required
  def show_directory(directory: str, page: int = 1):
      directory = unquote(directory)
      dir_path = _safe_path(_MEDIA_ROOT_R, directory)
      if not dir_path.is_dir():
          abort(404)
      mtime = dir_path.stat().st_mtime
      all_files, subdirs = _list_directory_cached(dir_path, mtime)
      total_pages = max(1, (len(all_files) + _PAGE_SIZE - 1) // _PAGE_SIZE)

      has_video = any(d.lower() == "video" for d in subdirs)
      video_files: list[str] = []
      if has_video:
          video_dir = dir_path / next(d for d in subdirs if d.lower() == "video")
          _VIDEO_EXTENSIONS = {".mp4", ".mov"}
          video_files = [
              f"/static/media/{directory}/{next(d for d in subdirs if d.lower() == 'video')}/{f.name}"
              for f in sorted(video_dir.iterdir())
              if f.is_file() and f.suffix.lower() in _VIDEO_EXTENSIONS
          ]

      return render_template(
          "directory.html",
          directory=directory,
          subdirs=subdirs if page == 1 else [],
          files=all_files[(page - 1) * _PAGE_SIZE : page * _PAGE_SIZE],
          total_pages=total_pages,
          current_page=page,
          has_video=has_video,
          video_files=video_files,
      )
  ```

## Task 10: JavaScript — Justified Rows

**Files:**
- Create: `static/js/justified-rows.js`

Lays out `.gallery-grid` thumbnails into justified rows where each row fills the full container width at a target height.

- [ ] **Step 1: Create static/js/justified-rows.js**

  ```js
  /* justified-rows.js — vanilla JS, no dependencies
   * Lays out .gallery-thumb elements in full-width rows at a target height.
   * Each thumb must have a data-ratio attribute (width/height, default 1.5).
   */
  (function () {
    var TARGET_HEIGHT = 220;
    var GAP = 4;

    function layout() {
      var grid = document.getElementById('galleryGrid');
      if (!grid) return;

      var thumbs = Array.prototype.slice.call(grid.querySelectorAll('.gallery-thumb'));
      if (!thumbs.length) return;

      var containerWidth = grid.clientWidth;
      if (!containerWidth) return;

      // Collect ratios
      var ratios = thumbs.map(function (el) {
        return parseFloat(el.getAttribute('data-ratio')) || 1.5;
      });

      // Determine target height based on screen width
      var targetH = window.innerWidth <= 600 ? 160 : TARGET_HEIGHT;

      var i = 0;
      while (i < thumbs.length) {
        // Find how many thumbs fit in one row
        var rowWidth = 0;
        var j = i;
        while (j < thumbs.length) {
          rowWidth += ratios[j] * targetH + (j > i ? GAP : 0);
          if (rowWidth > containerWidth && j > i) break;
          j++;
        }

        // Scale row to fill exactly containerWidth
        var rawWidth = 0;
        for (var k = i; k < j; k++) {
          rawWidth += ratios[k] * targetH;
        }
        var totalGap = (j - i - 1) * GAP;
        var scale = (containerWidth - totalGap) / rawWidth;
        var rowH = Math.round(targetH * scale);

        // Apply dimensions
        for (var k = i; k < j; k++) {
          var w = Math.round(ratios[k] * targetH * scale);
          thumbs[k].style.width = w + 'px';
          thumbs[k].style.height = rowH + 'px';
          var img = thumbs[k].querySelector('img');
          if (img) {
            img.style.width = w + 'px';
            img.style.height = rowH + 'px';
          }
        }

        i = j;
      }
    }

    // Debounce resize
    var _timer;
    function onResize() {
      clearTimeout(_timer);
      _timer = setTimeout(layout, 150);
    }

    document.addEventListener('DOMContentLoaded', layout);
    window.addEventListener('resize', onResize);
  })();
  ```

## Task 11: JavaScript — Lightbox

**Files:**
- Create: `static/js/lightbox.js`

Full-screen photo viewer with keyboard navigation, download, and touch swipe.

- [ ] **Step 1: Create static/js/lightbox.js**

  ```js
  /* lightbox.js — vanilla JS, no dependencies
   * Full-screen photo lightbox with keyboard nav, download, and touch swipe.
   * Expects: #lightbox, #lightboxImg, #lightboxCounter elements in the DOM.
   * Gallery thumbs must have .gallery-thumb and data-src attributes.
   */
  (function () {
    var _thumbs = [];
    var _current = 0;
    var _touchStartX = 0;

    function init() {
      _thumbs = Array.prototype.slice.call(
        document.querySelectorAll('.gallery-thumb')
      );
      if (!_thumbs.length) return;

      // Keyboard handler
      document.addEventListener('keydown', function (e) {
        var lb = document.getElementById('lightbox');
        if (!lb || !lb.classList.contains('open')) return;
        if (e.key === 'ArrowLeft')  moveLightbox(-1);
        if (e.key === 'ArrowRight') moveLightbox(1);
        if (e.key === 'Escape') closeLightbox();
        if (e.key === 'd' || e.key === 'D') downloadCurrent();
      });

      // Close on backdrop click (not on img/buttons)
      var lb = document.getElementById('lightbox');
      if (lb) {
        lb.addEventListener('click', function (e) {
          if (e.target === lb) closeLightbox();
        });
      }

      // Touch swipe
      if (lb) {
        lb.addEventListener('touchstart', function (e) {
          _touchStartX = e.changedTouches[0].clientX;
        }, { passive: true });
        lb.addEventListener('touchend', function (e) {
          var dx = e.changedTouches[0].clientX - _touchStartX;
          if (Math.abs(dx) > 50) moveLightbox(dx < 0 ? 1 : -1);
        }, { passive: true });
      }
    }

    function _show(index) {
      var lb  = document.getElementById('lightbox');
      var img = document.getElementById('lightboxImg');
      var ctr = document.getElementById('lightboxCounter');
      if (!lb || !img) return;

      _current = Math.max(0, Math.min(index, _thumbs.length - 1));
      var src = _thumbs[_current].getAttribute('data-src');
      img.src = src;
      if (ctr) ctr.textContent = (_current + 1) + ' / ' + _thumbs.length;
      lb.classList.add('open');
      document.body.style.overflow = 'hidden';

      // Preload adjacent
      if (_thumbs[_current + 1]) {
        new Image().src = _thumbs[_current + 1].getAttribute('data-src');
      }
    }

    // Exposed globally so onclick attributes in template can call them
    window.openLightbox = function (e, index) {
      e.preventDefault();
      _show(index);
    };

    window.closeLightbox = function () {
      var lb = document.getElementById('lightbox');
      if (lb) lb.classList.remove('open');
      document.body.style.overflow = '';
    };

    window.moveLightbox = function (dir) {
      _show(_current + dir);
    };

    window.downloadCurrent = function () {
      if (!_thumbs[_current]) return;
      var src = _thumbs[_current].getAttribute('data-src');
      var a = document.createElement('a');
      a.href = src;
      a.download = src.split('/').pop();
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
    };

    document.addEventListener('DOMContentLoaded', init);
  })();
  ```
